109  // gpio is tied to GND in order to wake up in HIGH

111  // Disable PULL_UP in order to allow it to wakeup on HIGH
112  rtc_gpio_pullup_dis((gpio_num_t)pin);
113  } break;
114

117  esp_sleep_enable_ext0_wakeup((gpio_num_t)pin, 0);
